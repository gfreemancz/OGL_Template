


find_library(SDL2_MAIN_LIBRARY
    NAMES libSDL2main    
    HINTS ${CMAKE_CURRENT_LIST_DIR}/lib
)

find_library(SDL2_LIBRARY
    NAMES libSDL2 
    HINTS ${CMAKE_CURRENT_LIST_DIR}/lib
)



list(APPEND EXTERNAL_LIBS_HEADER_PATHS "${CMAKE_CURRENT_LIST_DIR}/include/SDL2")

list(APPEND EXTERNAL_LIBS "${SDL2_MAIN_LIBRARY}")
list(APPEND EXTERNAL_LIBS "${SDL2_LIBRARY}")



set(EXTERNAL_LIBS_HEADER_PATHS "${EXTERNAL_LIBS_HEADER_PATHS}" PARENT_SCOPE)
set(EXTERNAL_LIBS "${EXTERNAL_LIBS}" PARENT_SCOPE)

message("SDL2")
